const fs = require('fs');

/**
 * Processa o conteúdo de um arquivo de dados de recrutamento.
 * @param {string} filePath - O caminho para o arquivo a ser processado.
 * @returns {object} - Um objeto contendo os resultados do processamento.
 * @throws {Error} - Lança um erro se o arquivo não for encontrado ou houver um erro de leitura.
 */
function processRecruitmentData(filePath) {
    if (!filePath) {
        throw new Error("Caminho do arquivo não fornecido.");
    }

    try {
        // 1. Leitura do arquivo
        const data = fs.readFileSync(filePath, 'utf8');
        
        // 2. Parsing dos dados
        // Separa as linhas, remove espaços em branco e filtra linhas vazias.
        const lines = data.trim().split(/\r?\n/);
        const validLines = lines
            .map(line => line.trim())
            .filter(line => line.length > 0);
        
        console.log(`[INFO] Arquivo lido com sucesso: ${filePath}`);
        console.log(`[INFO] Total de linhas válidas encontradas: ${validLines.length}`);

        // 3. Lógica de Processamento Otimizada (Exemplo de agregação)
        const recruitmentSummary = {
            totalEntries: validLines.length,
            // Aqui seria adicionada a lógica complexa de agregação,
            // como contagem de vagas, status, etc., baseada no conteúdo das linhas.
            statusCounts: {}
        };

        // Exemplo de agregação: Contar status (simulação)
        validLines.forEach(line => {
            if (line.includes('Status:')) {
                const status = line.split(':').slice(1).join(':').trim();
                recruitmentSummary.statusCounts[status] = (recruitmentSummary.statusCounts[status] || 0) + 1;
            }
        });

        return recruitmentSummary;

    } catch (error) {
        // Tratamento de erros mais específico
        if (error.code === 'ENOENT') {
            throw new Error(`ERRO: Arquivo não encontrado no caminho: ${filePath}`);
        } else {
            throw new Error(`ERRO FATAL durante o processamento do arquivo ${filePath}: ${error.message}`);
        }
    }
}
