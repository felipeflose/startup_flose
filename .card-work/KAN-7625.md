### 3. `services.py` (Business Logic)

```python
from typing import Dict, List
from models import ContentItem, GitSyncRequest, SyncResponse
from git_manager import GitManager
import uuid

class ContentService:
    def __init__(self, git_manager: GitManager):
        self.git_manager = git_manager
        # Simulated Database for content storage
        self._content_db: Dict[int, ContentItem] = {}
        self._next_id = 1

    def create_content(self, title: str, content: str) -> ContentItem:
        """Creates new content and prepares for Git integration.
        """
        new_id = self._next_id
        item = ContentItem(id=new_id, title=title, content=content, version=1, status="draft")
        self._content_db[new_id] = item
        self._next_id += 1

        print(f"Content created with ID: {new_id}")
        return item

    def sync_to_git(self, content_id: int, branch: str, message: str) -> SyncResponse:
        """Handles the synchronization of content to the Git workflow.
        """
        if content_id not in self._content_db:
            return SyncResponse(success=False, message="Content ID not found.")

        content = self._content_db[content_id]

        # 1. Update Version
        new_version = content.version + 1
        content.version = new_version

        # 2. Prepare Content for Git (Simulated)
        content_data = f"Title: {content.title}\nContent:\n{content.content}"
        
        # 3. Execute Git Commit via Manager
        git_hash = self.git_manager.commit_content(content_data, message)

        if git_hash:
            print(f"Successfully synced Content ID {content_id} to branch '{branch}' with hash: {git_hash}")
            return SyncResponse(success=True, message="Content successfully committed to Git.", new_version=new_version)
        else:
            # Rollback or error handling would happen here in a real scenario
            return SyncResponse(success=False, message="Git commit failed. Content remains in draft state.")

    def get_content(self, content_id: int) -> Optional[ContentItem]:
        """Retrieves content by ID."""
        return self._content_db.get(content_id)
