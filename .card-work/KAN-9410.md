import unittest
from unittest.mock import MagicMock
from src.gateway.api_handler import handle_request, MockGitService, MockArtifactGenerator

class TestApiHandler(unittest.TestCase):

    def setUp(self):
        # Configuração do Mock para simular serviços externos
        self.mock_git = MockGitService()
        self.mock_generator = MockArtifactGenerator()

    def test_generate_artifact_success(self):
        # Teste de fluxo de sucesso
        request = {'endpoint': '/generate_artifact', 'artifact_name': 'TestModel'}

        response = handle_request(request, self.mock_git, self.mock_generator)

        self.assertEqual(response['status'], 'success')
        self.assertIn('data', response)
        self.assertIn('Generated content', response['data'])

        # Verificação de interações com os mocks
        self.mock_git.fetch_code.assert_called_once_with('TestModel')
        self.mock_generator.generate.assert_called_once()

    def test_generate_artifact_git_failure(self):
        # Teste de falha ao buscar código no Git
        request = {'endpoint': '/generate_artifact', 'artifact_name': 'MissingModel'}

        # Mockando o Git para retornar código nulo
        self.mock_git.fetch_code = MagicMock(return_value=None)

        response = handle_request(request, self.mock_git, self.mock_generator)

        self.assertEqual(response['status'], 'error')
        self.assertIn('Source code for MissingModel not found in Git', response['message'])
        self.mock_generator.generate.assert_not_called()

    def test_generate_artifact_invalid_endpoint(self):
        # Teste de endpoint inválido
        request = {'endpoint': '/unknown'}

        response = handle_request(request, self.mock_git, self.mock_generator)

        self.assertEqual(response['status'], 'error')
        self.assertIn('Invalid endpoint', response['message'])

    def test_status_endpoint(self):
        # Teste de endpoint de status
        request = {'endpoint': '/status'}

        response = handle_request(request, self.mock_git, self.mock_generator)

        self.assertEqual(response['status'], 'ok')
