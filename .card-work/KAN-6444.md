## Solução de Engenharia
A implementação da autenticação de usuários será tratada através de um serviço centralizado (`AuthService`) que gerencia o registro, login e validação de tokens. Para garantir a segurança, utilizaremos *hashing* seguro para senhas (simulado aqui com `hashlib` em Python) e uma estrutura básica para gerenciamento de sessões.

**Estrutura da Solução:**
1. **`UserService`**: Simula o armazenamento dos usuários.
2. **`AuthService`**: Contém a lógica de autenticação (login/registro).
3. **Token Management**: Geração e validação de um token de sessão (simulado).

Este código foca na camada de serviço de autenticação, separando a preocupação da lógica de negócio do módulo de tarefas.

**Linguagem:** Python (padrão para demonstração de lógica de backend/serviços)

```python
import hashlib
import os
from typing import Dict, Optional

# --- Módulo de Simulação de Banco de Dados ---
class UserService:
    """Simula o armazenamento de usuários e senhas hasheadas."""
    def __init__(self):
        # Armazena: {username: {'password_hash': '...', 'user_id': 1, ...}}
        self.users = {}

    def add_user(self, username: str, password_hash: str) -> bool:
        """Registra um novo usuário."""
        if username in self.users:
            return False  # Usuário já existe
        self.users[username] = {
            'password_hash': password_hash,
            'user_id': len(self.users) + 1
        }
        return True

    def get_user_by_username(self, username: str) -> Optional[Dict]:
        """Busca os dados do usuário."""
        return self.users.get(username)

# --- Serviço de Autenticação Principal ---
class AuthService:
    """Gerencia a lógica de autenticação e tokens."""
    def __init__(self, user_service: UserService):
        self.user_service = user_service
        # Simulação de armazenamento de tokens ativos (em um sistema real, isso seria um DB ou Redis)
        self.active_sessions: Dict[str, int] = {}
        self.TOKEN_EXPIRATION_SECONDS = 3600 # 1 hora

    def hash_password(self, password: str) -> str:
        """Gera um hash seguro para a senha."""
        salt = os.urandom(16)
        # Usando PBKDF2 com sal e iterações para segurança robusta
        key = hashlib.pbkdf2_hmac('sha256', password.encode('utf-8'), salt, 100000)
        return salt.hex() + ':' + key.hex()

    def verify_password(self, stored_hash: str, provided_password: str) -> bool:
        """Verifica se a senha fornecida corresponde ao hash armazenado."""
        try:
            salt_hex, hash_hex = stored_hash.split(':')
            salt = bytes.fromhex(salt_hex)
            expected_key = hashlib.pbkdf2_hmac('sha256', provided_password.encode('utf-8'), salt, 100000)
            return expected_key.hex() == hash_hex
        except ValueError:
            # Hash mal formatado ou erro de parsing
            return False

    def register_user(self, username: str, password: str) -> Optional[str]:
        """Registra um novo usuário e retorna o hash da senha."""
        hashed_password = self.hash_password(password)
        if self.user_service.add_user(username, hashed_password):
            return hashed_password
        return None

    def login(self, username: str, password: str) -> Optional[str]:
        """Autentica o usuário e gera um token de sessão se for bem-sucedido."""
        user = self.user_service.get_user_by_username(username)
        if not user:
            return None  # Usuário não encontrado

        if self.verify_password(user['password_hash'], password):
            # Simulação de geração de um JWT/Token seguro
            session_token = os.urandom(32).hex() 
            self.active_sessions[session_token] = user['user_id']
            return session_token
        else:
            return None  # Senha incorreta

    def validate_session(self, token: str) -> Optional[int]:
        """Valida a existência e validade de um token de sessão."""
        if token in self.active_sessions:
            user_id = self.active_sessions[token]
            # Em um sistema real, checar expiração aqui
            return user_id
        return None

# --- Exemplo de Uso (Simulação da Integração)
if __name__ == '__main__':
    print("--- Inicializando Sistema de Autenticação ---")
    user_db = UserService()
    auth_service = AuthService(user_db)

    # 1. Registro de Usuário
    print("Registrando usuário 'testuser'...")
    hashed_pwd = auth_service.register_user("testuser", "securePassword123")
    if hashed_pwd:
        print(f"Registro bem-sucedido. Hash gerado: {hashed_pwd[:20]}...")

    # 2. Tentativa de Login
    print("Tentando login com credenciais válidas...")
    token = auth_service.login("testuser", "securePassword123")

    if token:
        print(f"Login bem-sucedido! Token gerado: {token[:20]}...")
        
        # 3. Validação da Sessão (Simulando acesso ao módulo de tarefas)
        print("Validando sessão...")
        user_id = auth_service.validate_session(token)
        if user_id:
            print(f"Sessão válida. Usuário ID: {user_id}. Acesso liberado ao Módulo de Tarefas.")
        else:
            print("Erro na validação da sessão.")
    else:
        print("Falha no login. Credenciais inválidas.")

    # 4. Tentativa de Login com Senha Inválida
    print("Tentando login com senha errada...")
    auth_service.login("testuser", "wrongPassword")

    # 5. Tentativa de Acesso sem Login
    print("Tentando validar sessão nula...")
    auth_service.validate_session("invalid-token-xyz")
```
