## Solução para KAN-6329: Geração Dinâmica de Templates de Tarefas Baseados em Perfil de Função

O desafio exige a implementação de uma camada de serviço que mapeie perfis de função (Roles) a templates de tarefas pré-definidos, permitindo a geração dinâmica de *templates* no contexto do Jira. 

Como desenvolvedor sênior, proponho uma arquitetura baseada em um Serviço de Geração de Templates, utilizando classes e um padrão de configuração para garantir escalabilidade e manutenibilidade.

### Arquitetura Proposta

1.  **`RoleBasedTemplateService`**: O serviço central responsável por carregar a lógica de mapeamento dos perfis para os templates.
2.  **`TemplateRepository`**: Uma camada simulada (que em produção seria um banco de dados ou um arquivo de configuração) que armazena o mapeamento dos roles e seus respectivos conteúdos de template.
3.  **`TaskGenerator`**: A interface pública que utiliza o serviço para gerar o conteúdo dinâmico da tarefa.

class TemplateRepository:
    """Simula o repositório de dados dos templates e perfis de função."""
    def __init__(self):
        # Mapeamento: Role -> [Template ID, Placeholder Fields]
        self._template_map = {
            "Developer": [
                {"id": "DEV_BUG", "description": "Bug report for development task. Needs steps and expected behavior."},
                {"id": "DEV_FEATURE", "description": "Feature implementation request. Must include acceptance criteria."}
            ],
            "QA_Tester": [
                {"id": "QA_TEST", "description": "Test plan creation. Focus on edge cases and boundary conditions."},
                {"id": "QA_BUG", "description": "Defect reporting process. Requires environment details and reproduction steps."}
            ],
            "Product_Owner": [
                {"id": "PO_REQ", "description": "New requirement specification. Focus on business value and scope definition."}
            ]
        }

    def get_templates_by_role(self, role: str) -> list[dict]:
        """Retorna a lista de templates disponíveis para um determinado papel."""
        return self._template_map.get(role, [])

    def get_template_details(self, template_id: str) -> dict:
        """Busca os detalhes completos de um template específico."""
        # Em uma implementação real, isso buscaria o conteúdo completo do DB.
        return {
            "DEV_BUG": {"title": "Bug Report", "template_text": "[Bug Report Template] - Define exact steps to reproduce and expected outcome."},
            "QA_TEST": {"title": "Test Case", "template_text": "[Test Case Template] - Setup environment and define success criteria."},
            "PO_REQ": {"title": "Requirement Spec", "template_text": "[Product Requirement Document] - Detail the business need and acceptance criteria."}
        }

class RoleBasedTaskService:
    """Serviço principal para gerenciar a geração de templates baseados em função."""
    def __init__(self, repository: TemplateRepository):
        self.repository = repository

    def generate_dynamic_template(self, user_role: str, template_type: str) -> dict:
        """Gera o objeto de template dinamicamente."""
        if user_role not in self.repository._template_map:
            raise ValueError(f"Role '{user_role}' não encontrada no sistema.")
        
        # 1. Encontra os IDs disponíveis para a função
        available_templates = self.repository.get_templates_by_role(user_role)
        
        if not available_templates:
            raise RuntimeError(f"Nenhum template encontrado para o papel: {user_role}.")

        # 2. Seleciona o template desejado (simulando a entrada do usuário)
        selected_template = next((t for t in available_templates if t['id'] == template_type), None)

        if not selected_template:
            raise ValueError(f"Template '{template_type}' não é válido para o papel {user_role}.")

        # 3. Busca os detalhes completos do template (rich content)
        details = self.repository.get_template_details(selected_template['id'])

        # 4. Geração do Template Final (Dynamic Content Assembly)
        generated_task = {
            "Jira_Issue_Type": "Story",  # Tipo padrão baseado na função
            "Template_Title": details.get("title", "Default Task") ,
            "Template_Content": details.get("template_text", "Error: Template not found.") ,
            "Source_Role": user_role,
            "Generated_Fields": {
                "Specific_Fields": [f"Field_{k}: {v}" for k, v in selected_template.items()]
            }
        }

        return generated_task
