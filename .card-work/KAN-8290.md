```python
import os
import subprocess
from typing import List, Dict

class GitWorkflowService:
    """
    Service responsável por automatizar tarefas de backend e integrar fluxos de trabalho do Git.
    Implementa boas práticas de tratamento de erros e integração com o sistema de arquivos/pipelines.
    """
    def __init__(self, repo_path: str = './'):
        """Inicializa o serviço com o caminho do repositório.
        Args:
            repo_path: O diretório raiz do repositório Git.
        """
        self.repo_path = os.path.abspath(repo_path)
        if not os.path.isdir(self.repo_path):
            raise FileNotFoundError(f"Diretório do repositório não encontrado: {repo_path}")

    def get_git_status(self) -> Dict[str, str]:
        """
        Executa 'git status' e retorna o resultado como um dicionário.
        """
        try:
            result = subprocess.run([
                'git', 'status', '--porcelain'
            ], cwd=self.repo_path, capture_output=True, text=True, check=True)
            
            statuses = {}
            for line in result.stdout.splitlines():
                if line:
                    # Exemplo de parsing simples: ' M file.txt'
                    status = line[:2].strip()
                    details = line[2:].strip()
                    statuses[details] = status
            return statuses
        except subprocess.CalledProcessError as e:
            raise RuntimeError(f"Erro ao executar git status: {e}")
        except FileNotFoundError:
            raise EnvironmentError("O comando 'git' não foi encontrado. Certifique-se de que o Git está instalado e no PATH.")

    def trigger_ci_pipeline(self, branch: str = 'main') -> bool:
        """
        Simula o acionamento de um pipeline de CI/CD baseado na branch atual.
        Em um ambiente real, isso chamaria uma API de Jenkins, GitHub Actions, etc.
        """
        print(f"[INFO] Iniciando trigger do pipeline para a branch: {branch}")
        
        # Simulação de chamada de pipeline
        try:
            # subprocess.run(['./scripts/trigger_pipeline.sh', branch], cwd=self.repo_path, check=True)
            print(f"[SUCCESS] Pipeline para '{branch}' iniciado com sucesso.")
            return True
        except subprocess.CalledProcessError as e:
            print(f"[ERROR] Falha ao iniciar o pipeline: {e}")
            return False

    def commit_and_tag(self, message: str, tag: str) -> bool:
        """
        Realiza um commit com uma mensagem e adiciona um tag Git.
        """
        try:
            # 1. Commit
            print(f"[INFO] Realizando commit: '{message}'")
            subprocess.run([
                'git', 'add', '.'
            ], cwd=self.repo_path, check=True)
            
            subprocess.run([
                'git', 'commit', '-m', message
            ], cwd=self.repo_path, check=True)

            # 2. Tag
            print(f"[INFO] Adicionando tag: {tag}")
            subprocess.run([
                'git', 'tag', tag
            ], cwd=self.repo_path, check=True)
            
            print("✅ Commit e Tag realizados com sucesso.")
            return True
        except subprocess.CalledProcessError as e:
            raise RuntimeError(f"Falha durante a operação Git (Commit/Tag): {e}")

if __name__ == '__main__':
    # Exemplo de uso simulado
    try:
        service = GitWorkflowService()
        print("\n--- 1. Status do Repositório ---")
        status = service.get_git_status()
        print(status)

        print("\n--- 2. Trigger CI ---")
        success = service.trigger_ci_pipeline()
        print(f"CI Status: {'SUCESSO' if success else 'FALHA'}")

        if success:
            new_commit_message = "feat: Implementação do serviço de automação KAN-8290"
            tag_name = "v1.0.0-automation"
            print("\n--- 3. Commit e Tag ---")
            service.commit_and_tag(new_commit_message, tag_name)

    except (RuntimeError, FileNotFoundError, EnvironmentError) as e:
        print(f"FATAL ERROR: O processo de automação falhou. Detalhes: {e}")
    ```
