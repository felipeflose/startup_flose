// assign_role_tasks.cjs (Versão Refatorada)

/**
 * Atribui tarefas a um determinado papel (role). 
 * Implementa tratamento de erros robusto para logging detalhado e re-lançamento de exceções específicas.
 * 
 * @param {string} roleId O ID do papel a ser atribuído.
 * @param {Array<Object>} tasks Um array das tarefas a serem atribuídas.
 * @returns {Promise<Object>} O resultado da atribuição.
 * @throws {Error} Lança um erro se a atribuição falhar.
 */
async function assign_role_tasks(roleId, tasks) {
  if (!roleId || !Array.isArray(tasks) || tasks.length === 0) {
    const errorMsg = "Argumentos inválidos fornecidos. Role ID e um array de tarefas válidos são obrigatórios.";
    console.error("assign_role_tasks: Argumentos inválidos.", { roleId, tasks });
    throw new Error(errorMsg);
  }

  try {
    // 1. Validação de entrada (já tratada acima, mas reforçada no fluxo)

    // 2. Lógica principal de atribuição (Simulação de operação assíncrona)
    console.log(`Iniciando atribuição de ${tasks.length} tarefas para o Role ID: ${roleId}`);

    // Simula a chamada externa que pode falhar
    const result = await makeApiCall(roleId, tasks);

    console.log(`Sucesso: Tarefas atribuídas com sucesso para o Role ID: ${roleId}.`);
    return result;
  } catch (error) {
    // 3. Tratamento específico de erros
    const errorMessage = `Falha crítica ao atribuir tarefas para o Role ID: ${roleId}. Detalhe: ${error.message || 'Erro desconhecido'}`; 
    
    // Log detalhado para o sistema de monitoramento
    console.error("ERRO CRÍTICO em assign_role_tasks:", errorMessage, {
      roleId,
      errorDetails: error.stack,
      context: "assign_role_tasks"
    });

    // Re-lançar um erro padronizado para que o chamador possa tratá-lo
    throw new Error(`Falha na atribuição de tarefas (Role ID: ${roleId}): ${error.message || 'Erro de serviço'}`);
  }
}
