import unittest
from src.task_manager import TaskManager

class TestTaskManager(unittest.TestCase):

    def setUp(self):
        "Configura o gerenciador de tarefas antes de cada teste.
        """
        self.manager = TaskManager()
        self.epic_id = "EPIC-123"

    def test_create_task_success(self):
        "Testa a criação bem-sucedida de uma tarefa.
        """
        title = "Implementar autenticação JWT"
        description = "Adicionar suporte a tokens JWT para login e validação de sessão."

        result = self.manager.create_task(self.epic_id, "Story", title, description)

        # Verificações básicas da estrutura de retorno
        self.assertIn("id", result)
        self.assertIn("epic_id", result)
        self.assertEqual(result["type"], "Story")
        self.assertEqual(result["title"], title)
        self.assertEqual(result["status"], "To Do")

        # Verificação da estrutura do ID gerado (simulação)
        self.assertTrue(result["id"].startswith("TASK-"))

    def test_create_task_missing_epic_id(self):
        "Testa o erro quando o epic_id está faltando.
        """
        with self.assertRaisesRegex(ValueError, "Epic ID e tipo de tarefa são obrigatórios."):
            self.manager.create_task("", "Bug", "Bug Title", "Bug Description")

    def test_create_task_missing_task_type(self):
        "Testa o erro quando o task_type está faltando.
        """
        with self.assertRaisesRegex(ValueError, "Epic ID e tipo de tarefa são obrigatórios."):
            self.manager.create_task(self.epic_id, "", "Title", "Description")

    def test_create_task_invalid_type(self):
        "Testa o erro quando o tipo de tarefa é inválido.
        """
        with self.assertRaisesRegex(ValueError, "Tipo de tarefa inválido."):
            self.manager.create_task(self.epic_id, "Feature", "Invalid Task", "Some description")

if __name__ == '__main__':
    unittest.main()
