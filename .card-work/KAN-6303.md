from abc import ABC, abstractmethod
from typing import Dict, Any, List

# --- Domain Models ---
class WorkflowEvent(ABC):
    """Base class for workflow events."""
    @abstractmethod
    def get_type(self) -> str:
        pass

class JiraIssueUpdate(WorkflowEvent):
    """Represents an update from a Jira ticket."""
    def __init__(self, issue_key: str, status: str, updated_by: str):
        self.issue_key = issue_key
        self.status = status
        self.updated_by = updated_by

    def get_type(self) -> str:
        return "JIRA_UPDATE"

class GitCommitEvent(WorkflowEvent):
    """Represents an update from a Git repository."""
    def __init__(self, commit_hash: str, branch: str, message: str):
        self.commit_hash = commit_hash
        self.branch = branch
        self.message = message

    def get_type(self) -> str:
        return "GIT_COMMIT"

# --- Orchestrator Interface (Abstraction for Testability) ---
class WorkflowOrchestrator(ABC):
    """Abstract interface for workflow orchestration.
    Ensures that specific implementations can be swapped easily."""
    @abstractmethod
    def process_event(self, event: WorkflowEvent) -> Dict[str, Any]:
        """Processes a single incoming event and determines the required workflow action."""
        pass

    @abstractmethod
    def execute_workflow(self, action: str, context: Dict[str, Any]) -> bool:
        """Executes a specific workflow action (e.g., create branch, update status)."""
        pass


# --- Concrete Implementation ---
class JiraGitOrchestrator(WorkflowOrchestrator):
    """Concrete implementation handling the synchronization logic."""
    def __init__(self, jira_api_client: Any, git_api_client: Any):
        self.jira_client = jira_api_client  # Dependency Injection
        self.git_client = git_api_client    # Dependency Injection

    def process_event(self, event: WorkflowEvent) -> Dict[str, Any]:
        """Main orchestration logic based on the event type."""
        event_type = event.get_type()
        print(f"Orchestrator received {event_type} event.")

        if event_type == "JIRA_UPDATE":
            return self._handle_jira_update(event)
        elif event_type == "GIT_COMMIT":
            return self._handle_git_commit(event)
        else:
            raise ValueError(f"Unknown event type: {event_type}")

    def _handle_jira_update(self, event: JiraIssueUpdate) -> Dict[str, Any]:
        """Logic for handling Jira updates."""
        print(f"-> Processing Jira update for {event.issue_key}: Status is {event.status}")
        # Example: Trigger a branch creation workflow if status moves to 'In Progress'
        if event.status == "In Progress":
            action = "CREATE_GIT_BRANCH"
            context = {
                "issue_key": event.issue_key,
                "new_branch": f"{event.issue_key}-in-progress",
                "user": event.updated_by
            }
            success = self.execute_workflow(action, context)
            return {"status": "Success", "message": f"Jira update processed. Workflow triggered: {action}. Success: {success}"}
        return {"status": "Info", "message": "No specific workflow action required for this status."}

    def _handle_git_commit(self, event: GitCommitEvent) -> Dict[str, Any]:
        """Logic for handling Git commits."""
        print(f"-> Processing Git commit {event.commit_hash} on branch {event.branch}")
        # Example: Log the commit to Jira if it relates to an open issue.
        action = "LOG_TO_JIRA"
        context = {
            "commit_sha": event.commit_hash,
            "ref": event.branch,
            "message": event.message
        }
        success = self.execute_workflow(action, context)
        return {"status": "Success", "message": f"Git commit processed. Workflow triggered: {action}. Success: {success}"}

    def execute_workflow(self, action: str, context: Dict[str, Any]) -> bool:
        """Simulated execution of a workflow step."""
        print(f"--- Executing Workflow Action: {action} with Context: {context} ---")
        # In a real scenario, this would call external services (Git/Jira APIs)
        if action == "CREATE_GIT_BRANCH":
            # Simulated success
            return True
        elif action == "LOG_TO_JIRA":
            # Simulated success
            return True
        else:
            print(f"Error: Unknown workflow action requested: {action}")
            return False


# --- Example Usage (Simulated Setup) ---
if __name__ == "example":

    class MockJiraAPI: pass
    class MockGitAPI: pass

    orchestrator = JiraGitOrchestrator(MockJiraAPI(), MockGitAPI())

    jira_event = JiraIssueUpdate(
        issue_key="PROJ-123", 
        status="In Progress", 
        updated_by="user@example.com"
    )
    result1 = orchestrator.process_event(jira_event)
    print(f"FINAL RESULT 1: {result1}")

    git_event = GitCommitEvent(

        branch="feature/new-api", 
        message="Implement initial API endpoints"
    )
    result2 = orchestrator.process_event(git_event)
