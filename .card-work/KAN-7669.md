import uuid
from datetime import datetime

class WorkItem:
    """Representa um Card de Trabalho (Work Item) no sistema.
    Implementa os atributos básicos e o método de inicialização.
    """
    def __init__(self, title: str, description: str, status: str = "To Do", priority: str = "Medium"): 
        if not title or not description:
            raise ValueError("Título e descrição são obrigatórios para a criação de um Work Item.")
            
        self.id = str(uuid.uuid4())
        self.title = title
        self.description = description
        self.status = status
        self.priority = priority
        self.created_at = datetime.now()
        self.updated_at = self.created_at

    def update(self, title: str = None, description: str = None, status: str = None, priority: str = None):
        """Atualiza os detalhes do Work Item.
        """
        if title is not None:
            self.title = title
        if description is not None:
            self.description = description
        if status is not None:
            self.status = status
        if priority is not None:
            self.priority = priority
            
        self.updated_at = datetime.now()

    def to_dict(self):
        """Retorna o Work Item em um formato de dicionário para serialização.
        """
        return {
            "id": self.id,
            "title": self.title,
            "description": self.description,
            "status": self.status,
            "priority": self.priority,
            "created_at": self.created_at.isoformat(),
            "updated_at": self.updated_at.isoformat()
        }

class WorkItemService:
    """Serviço responsável pela criação e gerenciamento de Work Items.
    """
    def __init__(self):
        # Simulação de um banco de dados em memória
        self._work_items = {}

    def create_work_item(self, title: str, description: str, status: str = "To Do", priority: str = "Medium") -> WorkItem:
        """Cria e inicializa um novo Work Item.
        """
        new_item = WorkItem(title=title, description=description, status=status, priority=priority)
        self._work_items[new_item.id] = new_item
        return new_item

    def get_work_item(self, item_id: str) -> WorkItem | None:
        """Recupera um Work Item pelo ID.
        """
        return self._work_items.get(item_id)

    def list_all_work_items(self) -> list[WorkItem]:
        """Retorna todos os Work Items registrados.
        """
        return list(self._work_items.values())

# Exemplo de Uso:
if __name__ == "__main__":
    service = WorkItemService()
    print("--- Criando Novo Card de Trabalho ---")
    try:
        item1 = service.create_work_item(
            title="Implementar Autenticação JWT",
            description="Configurar a camada de autenticação baseada em tokens JWT para todas as rotas críticas.",
            status="In Progress",
            priority="High"
        )
        print(f"Card criado com sucesso: {item1.to_dict()}")

        print("\n--- Listando Todos os Cards ---")
        all_items = service.list_all_work_items()
        for item in all_items:
            print(f"ID: {item.id[:8]}... | Título: {item.title} | Status: {item.status} | Prioridade: {item.priority}")
	
    except ValueError as e:
        print(f"Erro na criação: {e}")
