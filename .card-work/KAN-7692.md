### 2. `repositories/git_repo.py` (The Core Automation Logic)

This class encapsulates all interactions with the Git system, abstracting the complexity away from the service layer.

```python
import subprocess
from typing import List, Dict

class GitRepositoryManager:
    """Manages all interactions with the Git repository for workflow automation."""

    def __init__(self, repo_path: str):
        self.repo_path = repo_path

    def get_current_status(self) -> Dict[str, str]:
        """Retrieves the current status of the Git repository."""
        try:
            result = subprocess.run([
                'git', 'status', '--porcelain'
            ], cwd=self.repo_path, capture_output=True, text=True)
            if result.returncode != 0:
                raise RuntimeError(f"Git command failed: {result.stderr}")
            
            statuses = {}
            for line in result.stdout.splitlines():
                if line:
                    # Example parsing for status codes
                    status_code = line[:2]
                    statuses[line] = 'changed' if line else 'untracked'
            return statuses
        except FileNotFoundError:
            raise EnvironmentError("Git executable not found. Ensure Git is installed and in PATH.")

    def create_commit(self, message: str, content_path: str) -> str:
        """Creates a new commit based on linked content."""
        try:
            # Simulate linking by staging the content first
            subprocess.run(['git', 'add', content_path], cwd=self.repo_path, check=True)
            
            commit_message = f"[LINKED] {message}"
            subprocess.run(['git', 'commit', '-m', commit_message], cwd=self.repo_path, check=True)
            
            # Return the hash of the new commit for linking confirmation
            commit_hash = subprocess.check_output([
                'git', 'rev-parse', 'HEAD'
            ], cwd=self.repo_path).decode().strip()
            return commit_hash
        except subprocess.CalledProcessError as e:
            raise RuntimeError(f"Failed to create Git commit: {e}")

    def push_changes(self, branch: str) -> bool:
        """Pushes the latest changes to the remote repository."""
        try:
            subprocess.run(['git', 'push', '--set-upstream', 'origin', branch], cwd=self.repo_path, check=True)
            return True
        except subprocess.CalledProcessError as e:
            raise RuntimeError(f"Failed to push changes: {e}")

```

### 3. `core/services.py` (Business Logic Layer)

This layer orchestrates the workflow, handling validation and calling repository methods.

```python
from repositories.git_repo import GitRepositoryManager
from typing import Dict

class ContentLinkerService:
    """Handles the business logic for content linking and Git automation."""

    def __init__(self, git_manager: GitRepositoryManager):
        self.git_manager = git_manager

    def process_content_link(self, content_path: str, commit_message: str, branch_name: str) -> Dict:
        """Links a new file to the repository by creating a Git commit."""
        print(f"--- Starting link process for {content_path} ---")
        
        # 1. Check if content exists (simple validation)
        if not content_path.endswith('.md') and not content_path.endswith('.txt'):
            raise ValueError("Content path must be a supported file type (.md or .txt).")

        # 2. Perform Git Automation
        try:
            commit_hash = self.git_manager.create_commit(content_message, content_path)
            print(f"Successfully committed to {commit_hash}")
            
            # 3. Push the changes
            self.git_manager.push_changes(branch_name)
            print(f"Successfully pushed changes to branch: {branch_name}")
            
            return {
                "status": "success",
                "message": "Content linked and workflow automated successfully.",
                "commit_hash": commit_hash,
                "branch": branch_name
            }
        
        except (RuntimeError, ValueError) as e:
            print(f"Workflow Error: {e}")
            raise RuntimeError(f"Workflow failed during Git operation: {str(e)}")

    def get_repo_status(self) -> Dict:
        """Retrieves the current state of the repository."""
        return self.git_manager.get_current_status()

```

### 4. `api/endpoints.py` (API Layer)

This layer handles HTTP requests and maps them to the service logic.

```python
from fastapi import APIRouter, Depends, HTTPException, status
from typing import Dict

from core.services import ContentLinkerService
from repositories.git_repo import GitRepositoryManager
from models.schemas import LinkRequest, LinkResponse

# Dependency Injection Setup (In a real app, this would be managed by a DI container)
def get_link_service(manager: GitRepositoryManager) -> ContentLinkerService:
    """Provides the service layer instance."""
    return ContentLinkerService(git_manager=manager)

router = APIRouter()

@router.post("/link", response_model=LinkResponse, status_code=status.HTTP_201_CREATED)
def link_content_to_git( 
    request: LinkRequest,
    service: ContentLinkerService = Depends(get_link_service) 
) -> LinkResponse:
    """Endpoint to initiate content linking and Git workflow automation."""
    try:
        result = service.process_content_link(
            content_path=request.content_path,
            commit_message=request.commit_message,
            branch_name=request.branch
        )
        return LinkResponse(**result)
    except ValueError as ve:
        raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail=str(ve))
    except RuntimeError as re:
        # Catches errors from the repository layer (e.g., Git failures)
        raise HTTPException(status_code=status.HTTP_500_INTERNAL_SERVER_ERROR, detail=f"Workflow execution failed: {str(re)}")
